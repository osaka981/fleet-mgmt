// ==================================================== // 自動バックアップ（日次） // ==================================================== /** * Sheetを日付付きでバックアップフォルダにコピー（30日分保持） * GASエディタから setupBackupTrigger() を手動実行してトリガー設定 */ function dailyBackup() { try { var ss = SpreadsheetApp.openById(SHEET_ID); var today = Utilities.formatDate(new Date(), 'Asia/Tokyo', 'yyyy-MM-dd'); var backupName = 'fleet_backup_' + today; // バックアップフォルダを取得or作成（OUTPUT_FOLDER内） var parentFolder = DriveApp.getFolderById(OUTPUT_FOLDER_ID); var backupFolders = parentFolder.getFoldersByName('_バックアップ'); var backupFolder; if (backupFolders.hasNext()) { backupFolder = backupFolders.next(); } else { backupFolder = parentFolder.createFolder('_バックアップ'); } // 同日のバックアップがあればスキップ var existing = backupFolder.getFilesByName(backupName); if (existing.hasNext()) { Logger.log('本日のバックアップは既に存在します: ' + backupName); return; } // Sheetをコピー var copy = ss.copy(backupName); var copyFile = DriveApp.getFileById(copy.getId()); // バックアップフォルダに移動 backupFolder.addFile(copyFile); var parents = copyFile.getParents(); while (parents.hasNext()) { var parent = parents.next(); if (parent.getId() !== backupFolder.getId()) { parent.removeFile(copyFile); } } Logger.log('バックアップ完了: ' + backupName); // 古いバックアップを削除（30日以上前） cleanOldBackups(backupFolder); } catch(e) { Logger.log('バックアップエラー: ' + e.message); } } /** * 30日以上前のバックアップファイルを削除 */ function cleanOldBackups(backupFolder) { var cutoff = new Date(); cutoff.setDate(cutoff.getDate() - 30); var files = backupFolder.getFiles(); var deleted = 0; while (files.hasNext()) { var file = files.next(); if (file.getDateCreated() < cutoff) { file.setTrashed(true); deleted++; Logger.log('古いバックアップを削除: ' + file.getName()); } } if (deleted > 0) Logger.log(deleted + '件の古いバックアップを削除しました'); } /** * 毎朝6時にバックアップするトリガーを設定 * ★ GASエディタから手動で1回だけ実行してください */ function setupBackupTrigger() { // 既存のバックアップトリガーを削除 var trigs = ScriptApp.getProjectTriggers(); for (var i = 0; i < trigs.length; i++) { if (trigs[i].getHandlerFunction() === 'dailyBackup') { ScriptApp.deleteTrigger(trigs[i]); } } // 毎朝6時に実行（期限アラート9時の前） ScriptApp.newTrigger('dailyBackup') .timeBased() .atHour(6) .everyDays(1) .create(); Logger.log('バックアップトリガーを設定しました（毎朝6時）'); } /** * 手動バックアップ（テスト用） * GASエディタから実行して動作確認できます */ function manualBackup() { dailyBackup(); Logger.log('手動バックアップ完了'); }
